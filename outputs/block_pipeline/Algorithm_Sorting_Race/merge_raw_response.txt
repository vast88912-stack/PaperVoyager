// --- TYPES & INTERFACES ---

type Item = { val: number; id: number; originalIdx: number };
type Action = 
  | { type: 'COMPARE'; idx: [number, number] }
  | { type: 'SWAP'; idx: [number, number] }
  | { type: 'SET'; idx: number; item: Item }
  | { type: 'FINISH' };

type Algorithm = {
  id: string;
  name: string;
  run: (arr: Item[]) => Action[];
};

type Stats = {
  ops: number;
  swaps: number;
  finished: boolean;
};

type DatasetType = 'Random' | 'Nearly Sorted' | 'Reversed' | 'Few Uniques';
type Complexity = 'O(1)' | 'O(N)' | 'O(N log N)' | 'O(N²)' | 'O(N + K)' | 'O(N * K)';

  id: string;
  name: string;
  timeMs: number;
  operations: number;
  swaps: number;
  isStable: boolean;
  color: string;
}

interface AlgorithmDef {
  id: string;

  type: string;
  best: Complexity;
  average: Complexity;
  worst: Complexity;
  space: Complexity;
  stable: boolean;

  color: string;
}

// --- SORTING ALGORITHMS (PURE FUNCTIONS) ---

const bubbleSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  for (let i = 0; i < a.length; i++) {
    let swapped = false;
    for (let j = 0; j < a.length - i - 1; j++) {
      actions.push({ type: 'COMPARE', idx: [j, j + 1] });
      if (a[j].val > a[j + 1].val) {
        actions.push({ type: 'SWAP', idx: [j, j + 1] });
        [a[j], a[j + 1]] = [a[j + 1], a[j]];
        swapped = true;
      }
    }
    if (!swapped) break;
  }
  actions.push({ type: 'FINISH' });
  return actions;
};

const insertionSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  for (let i = 1; i < a.length; i++) {
    let j = i;
    while (j > 0) {
      actions.push({ type: 'COMPARE', idx: [j - 1, j] });
      if (a[j - 1].val > a[j].val) {
        actions.push({ type: 'SWAP', idx: [j - 1, j] });
        [a[j], a[j - 1]] = [a[j - 1], a[j]];
        j--;
      } else break;
    }
  }
  actions.push({ type: 'FINISH' });
  return actions;
};

const mergeSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  const merge = (l: number, m: number, r: number) => {
    const left = a.slice(l, m + 1);
    const right = a.slice(m + 1, r + 1);
    let i = 0, j = 0, k = l;
    while (i < left.length && j < right.length) {
      actions.push({ type: 'COMPARE', idx: [l + i, m + 1 + j] });
      if (left[i].val <= right[j].val) {
        actions.push({ type: 'SET', idx: k, item: left[i] });
        a[k++] = left[i++];
      } else {
        actions.push({ type: 'SET', idx: k, item: right[j] });
        a[k++] = right[j++];
      }
    }
    while (i < left.length) {
      actions.push({ type: 'SET', idx: k, item: left[i] });
      a[k++] = left[i++];
    }
    while (j < right.length) {
      actions.push({ type: 'SET', idx: k, item: right[j] });
      a[k++] = right[j++];
    }
  };
  const sort = (l: number, r: number) => {
    if (l >= r) return;
    const m = l + Math.floor((r - l) / 2);
    sort(l, m);
    sort(m + 1, r);
    merge(l, m, r);
  };
  sort(0, a.length - 1);
  actions.push({ type: 'FINISH' });
  return actions;
};

const quickSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  const partition = (low: number, high: number) => {
    const pivot = a[high].val;
    let i = low - 1;
    for (let j = low; j < high; j++) {
      actions.push({ type: 'COMPARE', idx: [j, high] });
      if (a[j].val < pivot) {
        i++;
        actions.push({ type: 'SWAP', idx: [i, j] });
        [a[i], a[j]] = [a[j], a[i]];
      }
    }
    actions.push({ type: 'SWAP', idx: [i + 1, high] });
    [a[i + 1], a[high]] = [a[high], a[i + 1]];
    return i + 1;
  };
  const sort = (low: number, high: number) => {
    if (low < high) {
      const pi = partition(low, high);
      sort(low, pi - 1);
      sort(pi + 1, high);
    }
  };
  sort(0, a.length - 1);
  actions.push({ type: 'FINISH' });
  return actions;
};

const heapSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  const heapify = (n: number, i: number) => {
    let largest = i;
    const l = 2 * i + 1;
    const r = 2 * i + 2;
    if (l < n) {
      actions.push({ type: 'COMPARE', idx: [l, largest] });
      if (a[l].val > a[largest].val) largest = l;
    }
    if (r < n) {
      actions.push({ type: 'COMPARE', idx: [r, largest] });
      if (a[r].val > a[largest].val) largest = r;
    }
    if (largest !== i) {
      actions.push({ type: 'SWAP', idx: [i, largest] });
      [a[i], a[largest]] = [a[largest], a[i]];
      heapify(n, largest);
    }
  };
  for (let i = Math.floor(a.length / 2) - 1; i >= 0; i--) heapify(a.length, i);
  for (let i = a.length - 1; i > 0; i--) {
    actions.push({ type: 'SWAP', idx: [0, i] });
    [a[0], a[i]] = [a[i], a[0]];
    heapify(i, 0);
  }
  actions.push({ type: 'FINISH' });
  return actions;
};

const radixSortRun = (arr: Item[]): Action[] => {
  const actions: Action[] = [];
  const a = [...arr];
  const max = Math.max(...a.map(x => x.val));
  for (let exp = 1; Math.floor(max / exp) > 0; exp *= 10) {
    const output = new Array(a.length).fill(null);
    const count = new Array(10).fill(0);
    for (let i = 0; i < a.length; i++) count[Math.floor(a[i].val / exp) % 10]++;
    for (let i = 1; i < 10; i++) count[i] += count[i - 1];
    for (let i = a.length - 1; i >= 0; i--) {
      const idx = Math.floor(a[i].val / exp) % 10;
      output[count[idx] - 1] = a[i];
      count[idx]--;
    }
    for (let i = 0; i < a.length; i++) {
      a[i] = output[i];
      actions.push({ type: 'SET', idx: i, item: a[i] });
    }
  }
  actions.push({ type: 'FINISH' });
  return actions;
};

const generateDataset = (type: DatasetType, size: number): Item[] => {
  let arr: number[] = [];
  if (type === 'Random') {
    arr = Array.from({ length: size }, () => Math.floor(Math.random() * 100) + 1);
  } else if (type === 'Nearly Sorted') {
    arr = Array.from({ length: size }, (_, i) => Math.floor((i / size) * 100) + 1);
    for (let i = 0; i < size / 10; i++) {
      const idx1 = Math.floor(Math.random() * size);
      const idx2 = Math.floor(Math.random() * size);
      [arr[idx1], arr[idx2]] = [arr[idx2], arr[idx1]];
    }
  } else if (type === 'Reversed') {
    arr = Array.from({ length: size }, (_, i) => 100 - Math.floor((i / size) * 100));
  } else if (type === 'Few Uniques') {
    const uniques = [10, 30, 50, 70, 90];
    arr = Array.from({ length: size }, () => uniques[Math.floor(Math.random() * uniques.length)]);
  }
  return arr.map((val, i) => ({ val, id: Math.random(), originalIdx: i }));
};

  const [speed, setSpeed] = useState<number>(50);
  const [selectedAlgos, setSelectedAlgos] = useState<string[]>(['bubble', 'merge', 'quick']);
  const [isRacing, setIsRacing] = useState(false);
  
  const [initialArray, setInitialArray] = useState<Item[]>([]);
  const [laneData, setLaneData] = useState<Record<string, { array: Item[], stats: Stats, activeIdx: number[] }>>({});
  
  const tracesRef = useRef<Record<string, Action[]>>({});
  const progressRef = useRef<Record<string, number>>({});
  const animationRef = useRef<number>();

  const initData = useCallback(() => {
    const arr = generateDataset(dataset, arraySize);
    setInitialArray(arr);
    const initialLaneData: any = {};
    selectedAlgos.forEach(id => {
      initialLaneData[id] = { array: [...arr], stats: { ops: 0, swaps: 0, finished: false }, activeIdx: [] };
    });
    setLaneData(initialLaneData);
    setIsRacing(false);
    if (animationRef.current) cancelAnimationFrame(animationRef.current);
  }, [dataset, arraySize, selectedAlgos]);

    initData();
  }, [initData]);

  const handleAlgoToggle = (id: string) => {
    setSelectedAlgos(prev => prev.includes(id) && prev.length > 1 ? prev.filter(a => a !== id) : prev.includes(id) ? prev : [...prev, id]);
  };

  const startRace = () => {
    if (isRacing) return;
    setIsRacing(true);
    
    const newLaneData: any = {};
    selectedAlgos.forEach(id => {
      const algo = ALGORITHMS.find(a => a.id === id);
      if (algo) {
        tracesRef.current[id] = algo.run([...initialArray]);
        progressRef.current[id] = 0;
        newLaneData[id] = { array: [...initialArray], stats: { ops: 0, swaps: 0, finished: false }, activeIdx: [] };
      }
    });
    setLaneData(newLaneData);

    let lastTime = performance.now();
    const tick = (time: number) => {
      const delay = 101 - speed;
      if (time - lastTime > delay) {
        lastTime = time;
        setLaneData(prev => {
          const next = { ...prev };
          let allFinished = true;
          selectedAlgos.forEach(id => {
            const trace = tracesRef.current[id];
            let p = progressRef.current[id];
            if (p < trace.length) {
              allFinished = false;
              const action = trace[p];
              const lane = { ...next[id], array: [...next[id].array], stats: { ...next[id].stats } };
              
              if (action.type === 'COMPARE') {
                lane.activeIdx = action.idx;
                lane.stats.ops++;
              } else if (action.type === 'SWAP') {
                lane.activeIdx = action.idx;
                const [i, j] = action.idx;
                [lane.array[i], lane.array[j]] = [lane.array[j], lane.array[i]];
                lane.stats.ops++;
                lane.stats.swaps++;
              } else if (action.type === 'SET') {
                lane.activeIdx = [action.idx];
                lane.array[action.idx] = action.item;
                lane.stats.ops++;
                lane.stats.swaps++;
              } else if (action.type === 'FINISH') {
                lane.activeIdx = [];
                lane.stats.finished = true;
              }
              
              next[id] = lane;
              progressRef.current[id]++;
            }
          });
          if (allFinished) {
            setIsRacing(false);
            return next;
          }
          return next;
        });
      }
      if (isRacing) animationRef.current = requestAnimationFrame(tick);
    };
    animationRef.current = requestAnimationFrame(tick);
  };

  return (
    <div className="p-6 md:p-12 max-w-7xl mx-auto space-y-8">
      <header className="flex items-center justify-between pb-6 border-b border-slate-800">
        <div>
          <h1 className="text-3xl font-extrabold tracking-tight text-slate-100 flex items-center gap-3">
            <Play className="w-8 h-8 text-indigo-500" /> Race Arena
          </h1>
          <p className="text-slate-400 mt-2 font-medium">Configure parameters and watch the algorithms compete.</p>
        </div>
      </header>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <div className="lg:col-span-2 space-y-6">
          <section className="bg-slate-900 p-6 rounded-2xl border border-slate-800">
            <h2 className="text-lg font-bold text-slate-100 mb-4 flex items-center gap-2">
              <Database className="w-5 h-5 text-indigo-500" /> Dataset Preset
            </h2>
            <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
              {DATASETS_DEF.map((opt) => (
                <button

                  onClick={() => !isRacing && setDataset(opt.name as DatasetType)}

                    dataset === opt.name

                      : 'border-slate-800 bg-slate-950 text-slate-400 hover:border-indigo-500/50 hover:bg-indigo-500/5'
                  } ${isRacing ? 'opacity-50 cursor-not-allowed' : ''}`}
                >
                  <span className="text-sm font-bold text-center leading-tight">{opt.name}</span>
                </button>
              ))}
            </div>
          </section>

          <section className="bg-slate-900 p-6 rounded-2xl border border-slate-800 space-y-8">
            <div>
              <div className="flex justify-between items-end mb-2">
                <label className="text-sm font-bold text-slate-300 uppercase tracking-wider">Array Size</label>
                <span className="text-xl font-black text-indigo-400 bg-indigo-500/10 px-3 py-1 rounded-lg">{arraySize}</span>
              </div>
              <input 
                type="range" min="10" max="200" value={arraySize}

                <div className="flex justify-center gap-4 min-h-[120px]">
                  {displayItems.map((item) => (
                    <div key={item.id} className={`relative flex flex-col items-center justify-center w-20 h-28 rounded-lg border-2 transition-all duration-500 ${item.color}`}>
                      <span className="text-3xl font-black">{item.key}</span>
                      <span className="absolute bottom-2 right-2 text-xs font-bold opacity-80">Type {item.variant}</span>
                    </div>
                  ))}
                </div>
              </div>
              <div className="flex flex-wrap gap-4 justify-center">
                <button onClick={() => setSortState('unsorted')} className={`px-6 py-2 font-bold text-sm rounded border transition-all ${sortState === 'unsorted' ? 'bg-slate-700 text-white border-slate-600' : 'bg-slate-950 text-slate-400 border-slate-800 hover:bg-slate-800'}`}>Reset (Unsorted)</button>
                <button onClick={() => setSortState('stable')} className={`px-6 py-2 font-bold text-sm rounded border transition-all ${sortState === 'stable' ? 'bg-emerald-600 text-white border-emerald-500' : 'bg-emerald-500/10 text-emerald-400 border-emerald-500/30 hover:bg-emerald-500/20'}`}>Sort (Stable)</button>
                <button onClick={() => setSortState('unstable')} className={`px-6 py-2 font-bold text-sm rounded border transition-all ${sortState === 'unstable' ? 'bg-rose-600 text-white border-rose-500' : 'bg-rose-500/10 text-rose-400 border-rose-500/30 hover:bg-rose-500/20'}`}>Sort (Unstable)</button>
              </div>
              {sortState === 'stable' && <div className="mt-6 p-4 bg-emerald-500/10 border-l-4 border-emerald-500 text-emerald-300 text-sm font-semibold rounded"><p>Notice how <strong>5 (Type A)</strong> still comes before <strong>5 (Type C)</strong>. Their relative order was preserved!</p></div>}
              {sortState === 'unstable' && <div className="mt-6 p-4 bg-rose-500/10 border-l-4 border-rose-500 text-rose-300 text-sm font-semibold rounded"><p>Notice how <strong>5 (Type C)</strong> now comes before <strong>5 (Type A)</strong>. The algorithm swapped them unnecessarily!</p></div>}
            </div>
          </div>
        )}
      </section>
    </div>
  );
};
